11 // If CUDA is available and in GPU mode, host memory will be allocated pinned,
12 // using cudaMallocHost. It avoids dynamic pinning for transfers (DMA).
13 // The improvement in performance seems negligible in the single GPU case,
14 // but might be more significant for parallel training. Most importantly,
15 // it improved stability for large models on many GPUs.
16 inline void CaffeMallocHost(void** ptr, size_t size, bool* use_cuda) {
17 #ifndef CPU_ONLY
18  if (Caffe::mode() == Caffe::GPU) {
19  CUDA_CHECK(cudaMallocHost(ptr, size));
20  *use_cuda = true;
21  return;
22  }
23 #endif
24  *ptr = malloc(size);
25  *use_cuda = false;
26  CHECK(*ptr) << "host allocation of size " << size << " failed";
27 }
